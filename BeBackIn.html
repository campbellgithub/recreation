<head>

    <style>

        body {

            font-family: Arial, sans-serif;

        }


        #message {

            font-size: 2em;

            margin-top: 20px;

        }

    </style>

</head>

<body>

    <h1 id="headline">Will Be Back in <span id="countdown">X</span> Minutes</h1>

    <p>Select minutes: 

        <select id="minutes">

            <!-- Options will be populated using JavaScript -->

        </select>

    </p>

    <button onclick="startCountdown()">Start Countdown</button>

    <p id="message"></p>


    <script>

        // Populate dropdown with options from 1 to 60

        let dropdown = document.getElementById('minutes');

        for(let i = 1; i <= 60; i++) {

            let option = document.createElement('option');

            option.text = i;

            option.value = i;

            dropdown.add(option);

        }


        function startCountdown() {

            let minutes = dropdown.value;

            let countdownElement = document.getElementById('countdown');

            countdownElement.innerText = minutes;


            let countdown = minutes * 60; // convert to seconds

            let intervalId = setInterval(function() {

                countdown -= 1;

                let remainingMinutes = Math.floor(countdown / 60);

                let remainingSeconds = countdown % 60;

                countdownElement.innerText = remainingMinutes + ":" + remainingSeconds;


                if(countdown <= 0) {

                    clearInterval(intervalId);

                    countdownElement.innerText = "X";

                    let message = document.getElementById('message');

                    message.innerText = "Will be back shortly following unforeseen and disadvantageous circumstances";

                }

            }, 1000);

        }

    </script>

</body>

</html>
